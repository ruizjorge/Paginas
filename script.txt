document.addEventListener("DOMContentLoaded", () => {
    renderMenu("menu-izquierdo", CONFIG.menuIzquierdo);
    renderMenu("menu-derecho", CONFIG.menuDerecho);
});

function renderMenu(menuId, grupos) {
    const menu = document.getElementById(menuId);

    grupos.forEach(grupo => {
        const contenedorGrupo = document.createElement("div");
        contenedorGrupo.className = "grupo";

        const titulo = document.createElement("h3");
        titulo.textContent = grupo.grupo;

        const lista = document.createElement("ul");

        grupo.enlaces.forEach(({ titulo, url }) => {
            const li = document.createElement("li");
            const a = document.createElement("a");

            a.href = url;
            a.target = "_blank";
            a.textContent = titulo;

            li.appendChild(a);
            lista.appendChild(li);
        });

        contenedorGrupo.appendChild(titulo);
        contenedorGrupo.appendChild(lista);
        menu.appendChild(contenedorGrupo);
    });
}
